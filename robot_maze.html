import random
import numpy as np

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3


def choose_action(epsilon):
    action = None

    randomprob = random.random()
#     print ('randomprob:',randomprob)
    if randomprob < epsilon: # 
        randomindex = random.randint(0, 3)
        action = actions[randomindex]# 
    else: 
        #  Q 
        action = max(qline, key=lambda x:qline.get(x))
        
#     print ('action:',action)
    return action
